\newcommand{\actor}[3]{

    \draw[thick] (-0.3,0.05) -- (0.3,0.05);

    \draw[thick] (0,-0.2) -- (0.2,-0.55);
    \node[actorlabel, below] at (0,-0.60) {#3};
  \end{scope}
}
